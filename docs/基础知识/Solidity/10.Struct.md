类似Java的Class对象

```solidity
// SPDX-License-Identifier: MIT
pragma solidity ^0.8.1;

contract Structs {
    struct Car {
        string model;
        uint256 year;
        address owner;
    }

    // 创建
    Car public car;
    Car[] public cars;

    mapping(address => Car[]) public carsByOwner;

    function examples() external {
        // 创建对象
        Car memory toyota = Car("Toyota", 1990, msg.sender);
        //创建对象
        Car memory lambo = Car({
            year: 1980,
            model: "Lamborghini",
            owner: msg.sender
        });
        // 创建对象
        Car memory tesla;
        
        // 更新信息
        tesla.model="Tesla";
        tesla.year=2010;
        tesla.owner=msg.sender;


        cars.push(toyota);
        cars.push(lambo);
        cars.push(tesla);

        cars.push(Car("Ferrari",2020,msg.sender));
    }
}
```
